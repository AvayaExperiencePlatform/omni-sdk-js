Interface AxpAudioInputInterface

AxpAudioInputInterface provides functionality for managing the audio input device that is used by the user (microphone).

interface AxpAudioInputInterface {
    addOnDevicesAddedCallback(callback): void;
    addOnDevicesRemovedCallback(callback): void;
    addOnDevicesUnavailableCallback(callback): void;
    addOnSelectedDeviceChangedCallback(callback): void;
    addOnSelectedDeviceRemovedCallback(callback): void;
    getAvailableDevices(): AxpMediaDevice[];
    getSelectedDevice(): AxpMediaDevice;
    getVolume(): number;
    hasAvailableDevices(): boolean;
    setSelectedDevice(device): void;
    setVolume(volume): any;
}
